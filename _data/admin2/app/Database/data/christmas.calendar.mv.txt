(function ($) {

    // year full; month 0-11; day: 1-31; hours: 0-23
    // for test purpose start day
    var startDay = Date.UTC(2022, 10, 1, 10);

    // final start day
    //  var startDay = Date.UTC(2022, 10, 29, 7);
    var endDay = Date.UTC(2022, 11, 30, 23);
    var today = Date.now();

    var fullDate = new Date();
    var todayDay = fullDate.getDate();

    var SAVE_TROPHY_ACTIVE_STATE = true;


    if (startDay <= today && today <= endDay) {
        $(".clue-box_day__back").css("visibility", "hidden");

        function setCookie(cName, cValue, exDays) {
            var day = new Date();
            day.setTime(day.getTime() + (exDays * 24 * 60 * 60 * 1000));
            var expires = "expires=" + day.toUTCString();
            document.cookie = cName + "=" + cValue + ";" + expires + ";path=/";
        }

        function getCookie(cName) {
            var name = cName + "=";
            var cValues = document.cookie.split(";");
            for (var i = 0; i < cValues.length; i++) {
                var cookie = cValues[i];
                while (cookie.charAt(0) == " ") {
                    cookie = cookie.substring(1);
                }
                if (cookie.indexOf(name) == 0) {
                    return cookie.substring(name.length, cookie.length);
                }
            }
            return "";
        };

        function inEditMode() {
            var queryString = window.location.search;
            var editContentString = "?editcontent";
            var inEditMode = queryString.indexOf(editContentString) > -1;

            return inEditMode;
        };


        function showOpenTrophy(index) {
            $("#calendar-button__id-" + index).css("display", "hidden");
            $("#clue-box-day__id-" + index).css("visibility", "visible");

            disableDoubleClick($("#clue-box-day__id-" + index));
        };

        function saveActiveTrophyIndexList(index) {
            var activeTrophy = (getCookie("activeTrophy") != "") ? JSON.parse(getCookie("activeTrophy")) : [];

            if (activeTrophy.indexOf(index) === -1)
                activeTrophy.push(index);

            setCookie("activeTrophy", JSON.stringify(activeTrophy), 300);
        };

        function setActiveTrophy() {
            var activeTrophy = (getCookie("activeTrophy") != "") ? JSON.parse(getCookie("activeTrophy")) : null;
            if (activeTrophy == null) {
                $("div.clue-section").hide();
                return;
            } else {
                $("div.clue-section").hide();
                $.each(activeTrophy, function (index, trophy) {
                    showOpenTrophy(trophy);
                    christmasClue(trophy);
                });

            }

        };

        function processToOpenTrophy(e) {
            var $clueBox = $(e.target);
            var $clueDayBoxIndex = $clueBox.closest(".christmas-calendar__clue-box_day").index();

            showOpenTrophy($clueDayBoxIndex);
            christmasClue($clueDayBoxIndex);

            if (SAVE_TROPHY_ACTIVE_STATE) {
                saveActiveTrophyIndexList($clueDayBoxIndex);
            }
        };

        // click to open today gift
        $(".christmas-calendar__clue-box__button_today").one("click", function (e) {
            e.preventDefault();
            var $todayDate = todayDay;
            var $todayIndex = $todayDate - 1;

            showOpenTrophy($todayIndex);
            christmasClue($todayIndex);

            if (SAVE_TROPHY_ACTIVE_STATE) {
                saveActiveTrophyIndexList($todayIndex);
            }

            disableDoubleClick($(this));
            e.preventDefault();

        });


        function christmasClue($index) {
            var $prevClue = $("div.clue-section-default");
            var $currentSelection = $("div#clue-section__id-" + $index);
            $prevClue.html($currentSelection.html());
        };


        function setEventListeners(index) {
            $("#calendar-button__id-" + index).on("click", processToOpenTrophy);
        };


        for (var i = 0; i < $(".christmas-calendar__clue-box_day").length; i++) {
            // calendar day
            var $calendarDayBox = $(".christmas-calendar__clue-box_day")[i];
            $($($calendarDayBox).find(".calender-button__clue-box_day")[0]).attr("id", "calendar-button__id-" + i);
            $($($calendarDayBox).find(".clue-box_day__back")[0]).attr("id", "clue-box-day__id-" + i);

            // clue day
            var $clueDayBox = $(".clue-section")[i];
            $($($clueDayBox)).attr("id", "clue-section__id-" + i);
        }
        ;


        function setAllVisibility() {
            $("body").addClass("show-all");

            $(".clue-box_day__back").css("visibility", "visible");

            $("div.clue-section-default").show();
            $("div.clue-section").show();
        };

        function disableDoubleClick($this) {
            $($this).prop("disabled", true);
            $($this).css({"cursor": "not-allowed"});
        };


        function makeClueBoxClickable() {
            var $todayDate = todayDay;
            var $todayIndex = $todayDate - 1;

            $(".christmas-calendar__clue-box_day").each(function (index, value) {
                if (index <= $todayIndex) {
                    setEventListeners(index);
                }
                if (index > $todayIndex) {
                    $("#clue-box-day__id-" + index).remove();
                    $("#clue-section__id-" + index).remove();

                    disableDoubleClick($("#calendar-button__id-" + index));
                }
            })

        };

        function main() {
            if (inEditMode()) {
                return setAllVisibility();
            }

            if (SAVE_TROPHY_ACTIVE_STATE) {
                setActiveTrophy();
            }

            makeClueBoxClickable();
        };

        main();
    } else {
        $("div.clue-section").hide();
        $(".clue-box_day__back").css("visibility", "hidden");
    }

})(jQuery);
